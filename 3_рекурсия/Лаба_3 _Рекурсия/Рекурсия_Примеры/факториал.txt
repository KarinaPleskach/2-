uses

var
  n,i: Integer;
  f: real;

\\Подсчет факториала с помощью рекурсии
Function fact (n:Integer) : real;
begin
  if n=0 then
    fact:=1
  else
    fact:=n*fact(n-1);
end;

\\Подсчет факториала без рекурсии 

function ff (n:integer):real;
begin
  Result:=1;
  for i:=2 to n do
    Result:=Result*i;
end;


begin
  writeln('fact ',10,'= ',fact(10):1:0);
  
  writeln('fact ',10,'= ',ff(10):1:0);
  readln;
end.
